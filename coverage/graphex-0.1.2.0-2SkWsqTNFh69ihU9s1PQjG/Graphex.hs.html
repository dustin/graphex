   36 -- | Convert a dependency file to a graph.
   37 depToGraph :: GraphDef -> Graph Text
   38 depToGraph GraphDef{..} = Graph (links <> allNodes) mappedAttrs
   39     where
   40         links = Map.fromListWith (<>) [ (name from, Set.singleton (name to)) | Edge{..} <- edges]
   41         allNodes = Map.fromList [(label, mempty) | Node{..} <- Map.elems nodes]
   42         name = (.label) . (nodes Map.!) :: Text -> Text
   43         mappedAttrs = Map.mapKeys name attrs
   44 
   45 -- | Convert a graph back to our dependency file format.
   46 graphToDep :: Graph Text -> GraphDef
   47 graphToDep (Graph m attrs) = GraphDef {
   48     title = "Internal Package Dependencies",
   49     edges = [ Edge { from = k, to = v } | (k, vs) <- Map.assocs m, v <- Set.toList vs ],
   50     nodes = Map.fromList [ (k, Node k Nothing) | k <- Map.keys m ],
   51     attrs = Map.fromList [ (k, v) | k <- Map.keys m, v <- maybe [] pure (Map.lookup k attrs)]
   52     }

   54 -- | Load a graph from a lookingglass JSON representation.
   55 getInput :: FilePath -> IO (Graph Text)
   56 getInput fn = either fail (pure . depToGraph) . eitherDecode =<< BL.readFile fn
   57 
   58 hGetInput :: Handle -> IO (Graph Text)
   59 hGetInput h = either fail (pure . depToGraph) . eitherDecode =<< BL.hGetContents h

   61 -- | Reverse all the arrows in the graphs.
   62 reverseEdges :: Ord a => Graph a -> Graph a
   63 reverseEdges g@(Graph m _) = g{unGraph = Map.fromListWith (<>) [ (v, Set.singleton k) | (k, vs) <- Map.assocs m, v <- Set.toList vs ] <> Map.fromSet (const mempty) (Map.keysSet m)}
   64 
   65 -- | Find the direct list of things that are referencing this module.
   66 directDepsOn :: Ord a => Graph a -> a -> Set a
   67 directDepsOn = flip (Map.findWithDefault mempty) . unGraph
   68 
   69 -- | Flood fill to find all transitive dependencies on a starting module, excluding the original key.
   70 allDepsOn :: Ord a => Graph a -> a -> Set a
   71 allDepsOn = flood . directDepsOn
   72 
   73 -- | Find an example path between two modules.
   74 --
   75 -- This is a short path, but the important part is that it represents how connectivity works.
   76 why :: Ord a => Graph a -> a -> a -> Maybe (NonEmpty a)
   77 why m from to = nonEmpty =<< findFirst bfsOn head (\ks -> (:ks) <$> (Set.toList . directDepsOn m . head) ks) [from] ((== to) . head)
   78 
   79 -- | Find all paths between two modules as a restricted graph of the intersection of
   80 -- reachable nodes from the start and to the end.
   81 allPathsTo :: Ord a => Graph a -> a -> a -> Graph a
   82 allPathsTo m from to = restrictTo m $ allDepsOn m from `Set.intersection` allDepsOn (reverseEdges m) to
   83 
   84 -- | Count the number of transitive dependencies for each module.
   85 rankings :: (NFData a, Ord a) => Graph a -> [(Int, a)]
   86 rankings g = sortOn (first Down) . fmap swap $ mapMaybeWithKey (Just . length . allDepsOn g) g
   87 
   88 -- | Visit each node in the graph and apply a function to it.
   89 mapMaybeWithKey :: (NFData a, NFData g) => (g -> Maybe a) -> Graph g -> [(g, a)]
   90 mapMaybeWithKey f = mapMaybe sequenceA . parMap rdeepseq (ap (,) f) . Map.keys . unGraph
   91 
   92 -- | Restrict a graph to only the given set of modules.
   93 restrictTo :: Ord a => Graph a -> Set a -> Graph a
   94 restrictTo (Graph m attrs) keep = Graph (flip Map.mapMaybeWithKey m $ \k' v ->
   95   if Set.notMember k' keep then Nothing else Just (Set.intersection keep v)) (Map.restrictKeys attrs keep)
   96 
   97 -- | Convert a Graph to a Tree. If there is a cycle, treat the cycle point as a leaf.
   98 graphToTree :: Ord a => a -> Graph a -> Tree a
   99 graphToTree startingAt (Graph rels _) = go mempty startingAt
  100   where
  101     go seen node =
  102       let children = Set.toList $ Map.findWithDefault mempty node rels
  103       in Tree.Node node $ if Set.member node seen then [] else fmap (go (Set.insert node seen)) children
  104 
  105 -- | Find the longest path in a Graph.
  106 longest :: (NFData a, Ord a) => Graph a -> [a]
  107 longest g = reverse . snd . maximumOn (length . snd) $ mapMaybeWithKey one g
  108   where
  109     one = Just . fmap fst . maximumOn length . dfsWith store seen (\ks -> (:ks) . (, succ . snd . head $ ks) <$> foldMap (Set.toList . directDepsOn g . fst) (listToMaybe ks)) . (:[]) . (,1::Int)
  110     -- store and head cannot be called with empty lists.
  111     store (head -> (k,l)) = Map.insert k l
  112     seen (head -> (k,l)) = maybe False (< l) . Map.lookup k
  113     maximumOn :: Ord b => (a -> b) -> [a] -> a
  114     maximumOn f = maximumBy (comparing f)
  115 
  116 filterNodes :: Ord a => (a -> Bool) -> Graph a -> Graph a
  117 filterNodes shouldRemove g =
  118   let shouldKeep = not . shouldRemove
  119       removeNodes (k, vs) = do
  120         guard $ shouldKeep k
  121         pure (k, Set.filter shouldKeep vs)
  122   in Graph
  123   { unGraph = Map.fromList $ mapMaybe removeNodes $ Map.toList $ unGraph g
  124   , attributes = Map.filterWithKey (\k _ -> shouldKeep k) (attributes g)
  125   }
