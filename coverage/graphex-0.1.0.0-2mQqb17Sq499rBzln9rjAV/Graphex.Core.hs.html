   12 data Graph a = Graph {
   13   unGraph    :: Map a (Set a),
   14   attributes :: Map a (Map Text Text)
   15   }
   16   deriving stock Eq
   17 
   18 instance Ord a => Semigroup (Graph a) where
   19   Graph x xa <> Graph y ya = Graph (Map.unionWith (<>) x y) (Map.unionWith (<>) xa ya)
   20 
   21 instance Ord a => Monoid (Graph a) where
   22   mempty = Graph mempty mempty
   23 
   24 singletonGraph :: Ord a => a -> a -> Graph a
   25 singletonGraph k = mkGraph k . pure
   26 
   27 mkGraph :: Ord a => a -> [a] -> Graph a
   28 mkGraph k links = Graph (Map.singleton k (Set.fromList links)) mempty
   29 
   30 setAttribute :: Ord a => a -> Text -> Text -> Graph a -> Graph a
   31 setAttribute k attr val g@(Graph _ attrs) = g{attributes = Map.insertWith (<>) k (Map.singleton attr val) attrs}
   32 
   33 getAttribute :: Ord a => a -> Text -> Graph a -> Maybe Text
   34 getAttribute nodeName aName = (Map.lookup aName <=< Map.lookup nodeName) . attributes
   35 
   36 graphNodes :: Graph a -> [a]
   37 graphNodes = Map.keys . unGraph

   64 singletonModuleGraph = singletonGraph

   69 -- | Convert a Graph of type @a@ to a Graph of type @b@.
   70 -- This is like a functor, but it can't be a functor because of laws and stuff.
   71 convertGraph :: Ord b => (a -> b) -> Graph a -> Graph b
   72 convertGraph f (Graph m attrs) = g{attributes = Map.mapKeys f attrs}
   73     where g = foldMap (\(k, vs) -> Graph (Map.singleton (f k) (Set.map f vs)) mempty) (Map.assocs m)
