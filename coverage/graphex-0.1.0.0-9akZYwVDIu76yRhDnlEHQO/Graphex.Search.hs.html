    3 module Graphex.Search (bfsOn, bfsWith, dfsOn, dfsWith, findFirst, flood) where

    5 import           Data.List     (find)
    6 import           Data.Set      (Set)
    7 import qualified Data.Set      as Set
    8 
    9 import           Graphex.Queue
   10 
   11 type SearchFunction r a = (a -> r) -> (a -> [a]) -> a -> [a]
   12 
   13 -- | Find the first match using the given search function (e.g., 'bfsOn' or 'dfsOn').
   14 findFirst :: Ord r => SearchFunction r a -> (a -> r) -> (a -> [a]) -> a -> (a -> Bool) -> Maybe a
   15 findFirst search rep nf start termination = find termination (search rep nf start)
   16 
   17 -- | A BFS that will return a list of all reachable states.
   18 --
   19 -- The first argument is a representation function is used to deduplicate state.
   20 -- For some use cases where the entire state is valid, you can use 'id'.
   21 bfsOn :: Ord r => (a -> r) -> (a -> [a]) -> a -> [a]
   22 bfsOn rep = bfsWith (Set.insert . rep) (Set.member . rep)
   23 
   24 -- | BFS with custom functions for remembering and recalling whether a state has been visited.
   25 bfsWith :: Monoid s => (a -> s -> s) -> (a -> s -> Bool) -> (a -> [a]) -> a -> [a]
   26 bfsWith remember seenf next start = go mempty (qsingle start)
   27   where
   28     go seen inq =
   29       case qpop inq of
   30         Nothing -> []
   31         Just (x,xs)
   32           | seenf x seen -> go seen xs
   33           | otherwise    -> x : go (remember x seen) (qappendList xs $ next x)
   34 
   35 -- | A DFS variant of 'bfsOn'.
   36 dfsOn :: Ord r => (a -> r) -> (a -> [a]) -> a -> [a]
   37 dfsOn rep = dfsWith (Set.insert . rep) (Set.member . rep)
   38 
   39 -- | A DFS variant of 'bfsWith'.
   40 dfsWith :: Monoid s => (a -> s -> s) -> (a -> s -> Bool) -> (a -> [a]) -> a -> [a]
   41 dfsWith remember seenf next start = go mempty [start]
   42   where
   43     go _ [] = []
   44     go seen (x:xs)
   45         | seenf x seen = go seen xs
   46         | otherwise    = x : go (remember x seen) (next x <> xs)
   47 
   48 -- | Flood fill a graph from a starting point and return all visited points.
   49 flood :: Ord a => (a -> Set a) -> a -> Set a
   50 flood nf = go mempty . Set.singleton
   51     where
   52         go !s (flip Set.difference s -> todo)
   53             | Set.null todo = s
   54             | otherwise = go (s <> todo) (Set.unions $ Set.map nf todo)
