   18   } deriving stock (Show, Generic)

   21 type NodeId = Text

   23   { from :: NodeId

   28 data GraphDef = GraphDef
   29   { title :: Text
   30   , nodes :: Map NodeId Node
   31   , edges :: [Edge]
   32   , attrs :: Map NodeId (Map Text Text)
   33   } deriving (Show, Generic)
   34   deriving anyclass Ae.ToJSON
   35 
   36 instance Ae.FromJSON GraphDef where
   37   parseJSON = withObject "GraphDef" $ \o -> do
   38     title <- o .: "title"
   39     nodes <- o .: "nodes"
   40     edges <- o .: "edges"
   41     attrs <- o .:? "attrs" .!= mempty
   42     pure GraphDef{..}

   54 toLookingGlass
   55   :: Text -- ^ title
   56   -> Map Text Color -- ^ Optionally re-color any nodes (black default)
   57   -> Graph Text
   58   -> GraphDef
   59 toLookingGlass title colors Graph{..} =
   60   let mkNodeId m = m
   61       mkNode m =
   62         ( mkNodeId m
   63         , Node
   64           { color = unColor <$> Map.lookup m colors
   65           , label = mkNodeId m
   66           }
   67         )
   68   in GraphDef
   69      { nodes = Map.fromList $ concatMap (\(m, children) -> mkNode m : fmap mkNode (Set.toList children) ) $ Map.toList unGraph
   70      , edges = Map.toList unGraph >>= \(m, children) -> fmap (\c -> Edge{to = mkNodeId c, from = mkNodeId m}) $ Set.toList children
   71      , attrs = attributes
   72      , ..
   73      }
