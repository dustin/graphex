   74 discoverCabalModules :: CabalDiscoverOpts -> FilePath -> IO [Module]
   75 discoverCabalModules CabalDiscoverOpts{..} cabalFile = do
   76   gpd <- readGenericPackageDescription silent cabalFile
   77   let PackageDescription{..} = flattenPackageDescription gpd
   78   let candidateModules = mconcat
   79         [ do
   80             Library{..} <- mconcat [maybeToList library, subLibraries]
   81             srcDir <- hsSourceDirs libBuildInfo
   82             exMod <- exposedModules
   83             let name = unUnqualComponentName <$> libraryNameString libName
   84             guard $ Discovered == foldMap1 (`discoversUnit` CabalLibraryUnit name) toDiscover
   85             pure Module
   86               { name = fromString $ mconcat $ intersperse "." $ Cabal.components exMod
   87               , path = ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath exMod <.> ".hs"
   88               }
   89         , do
   90             Executable{..} <- executables
   91             srcDir <- hsSourceDirs buildInfo
   92             otherMod <- "Main" : buildInfo.otherModules
   93             guard $ Discovered == foldMap1 (`discoversUnit` CabalExecutableUnit (unUnqualComponentName exeName)) toDiscover
   94             pure Module
   95               { name = fromString $
   96                 if otherMod == "Main"
   97                 then unUnqualComponentName exeName ++ "-Main"
   98                 else mconcat $ intersperse "." $ Cabal.components otherMod
   99               , path = ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath otherMod <.> ".hs"
  100               }
  101         , do
  102             TestSuite{..} <- testSuites
  103             srcDir <- hsSourceDirs testBuildInfo
  104             otherMod <- testBuildInfo.otherModules
  105             guard $ Discovered == foldMap1 (`discoversUnit` CabalTestsUnit (unUnqualComponentName testName)) toDiscover
  106 
  107             pure Module
  108               { name = fromString $ mconcat $ intersperse "." $ Cabal.components otherMod
  109               , path = ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath otherMod <.> ".hs"
  110               }
  111         ]
  112 
  113   pooledMapConcurrently validateModulePath candidateModules

  115 validateModulePath :: Module -> IO Module
  116 validateModulePath m = do
  117   path <- case m.path of
  118     ModuleFile fp -> do
  119       fileExists <- doesFileExist fp
  120       pure $ if fileExists then ModuleFile fp else ModuleNoFile
  121     ModuleNoFile -> pure ModuleNoFile
  122   pure Module {name = m.name, path = path}

  124 data CabalUnitType =
  125     CabalLibrary
  126   | CabalExecutable
  127   | CabalTests
  128   deriving stock (Show, Eq, Ord)
  129 
  130 data CabalUnit =
  131     CabalLibraryUnit (Maybe String)
  132   | CabalExecutableUnit String
  133   | CabalTestsUnit String
  134   deriving stock (Show, Eq, Ord)
  135 
  136 data CabalDiscoverType =
  137     CabalDiscoverAll CabalUnitType
  138   | CabalDiscover CabalUnit
  139   | CabalDontDiscoverAll CabalUnitType
  140   | CabalDontDiscover CabalUnit
  141   deriving stock (Show, Eq, Ord)
  142 
  143 data Discovery =
  144     Discovered
  145   | Hidden
  146   | Passed
  147   deriving stock (Show, Eq, Ord)
  148 
  149 instance Semigroup Discovery where
  150   _ <> Discovered = Discovered
  151   _ <> Hidden     = Hidden
  152   x <> Passed     = x
  153 
  154 flipDiscover :: Discovery -> Discovery
  155 flipDiscover = \case
  156   Passed -> Passed
  157   Hidden -> Discovered
  158   Discovered -> Hidden
  159 
  160 discoversUnit :: CabalDiscoverType -> CabalUnit -> Discovery
  161 discoversUnit = curry $ \case
  162   (CabalDiscoverAll CabalLibrary, CabalLibraryUnit{}) -> Discovered
  163   (CabalDiscoverAll CabalLibrary, _) -> Passed
  164   (CabalDiscoverAll CabalExecutable, CabalExecutableUnit{}) -> Discovered
  165   (CabalDiscoverAll CabalExecutable, _) -> Passed
  166   (CabalDiscoverAll CabalTests, CabalTestsUnit{}) -> Discovered
  167   (CabalDiscoverAll CabalTests, _) -> Passed
  168   (CabalDiscover u1, u2) -> if u1 == u2 then Discovered else Passed
  169   (CabalDontDiscoverAll ty, u) -> flipDiscover $ CabalDiscoverAll ty `discoversUnit` u
  170   (CabalDontDiscover u1, u2) -> flipDiscover $ CabalDiscover u1 `discoversUnit` u2
  171 
  172 data CabalDiscoverOpts = CabalDiscoverOpts
  173   { toDiscover      :: NonEmpty CabalDiscoverType
  174   , includeExternal :: Bool
  175   } deriving stock (Show, Eq)

  177 discoverCabalModuleGraph :: CabalDiscoverOpts -> IO ModuleGraph
  178 discoverCabalModuleGraph opts@CabalDiscoverOpts{..} = do
  179   fs <- getDirectoryContents "." -- XXX
  180   mods <- fmap fold . pooledMapConcurrently (discoverCabalModules opts) . filter ((".cabal" ==) . takeExtension) $ fs
  181 
  182   let modSet = foldMap (Set.singleton . name) mods
  183   gs <- pooledForConcurrently mods $ \Module{..} -> case path of
  184     ModuleFile modPath -> do
  185       allImps <- parseFileImports modPath
  186       let filteredImps =
  187             if includeExternal
  188             then allImps
  189             else filter (\Import{..} -> Set.member module_ modSet) allImps
  190       pure $ mkModuleGraph name $ fmap module_ filteredImps
  191     ModuleNoFile -> pure $ mkModuleGraph name mempty
  192   pure $ fold gs
