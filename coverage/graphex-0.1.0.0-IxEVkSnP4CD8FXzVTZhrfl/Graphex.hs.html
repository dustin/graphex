   22 -- | Convert a dependency file to a graph.
   23 depToGraph :: GraphDef -> Graph
   24 depToGraph GraphDef{..} = Graph (links <> allNodes)
   25     where
   26         links = Map.fromListWith (<>) [ (name from, Set.singleton (name to)) | Edge{..} <- edges]
   27         allNodes = Map.fromList [(label, mempty) | Node{..} <- Map.elems nodes]
   28         name = (.label) . (nodes Map.!) :: Text -> Text

   30 -- | Convert a graph back to our dependency file format.
   31 graphToDep :: Graph -> GraphDef
   32 graphToDep (Graph m) = GraphDef {
   33     title = "Internal Package Dependencies",
   34     edges = [ Edge { from = k, to = v } | (k, vs) <- Map.assocs m, v <- Set.toList vs ],
   35     nodes = Map.fromList [ (k, Node k Nothing) | k <- Map.keys m ]
   36     }

   38 -- | Reverse all the arrows in the graphs.
   39 reverseEdges :: Graph -> Graph
   40 reverseEdges (Graph m) = Graph $ Map.fromListWith (<>) [ (v, Set.singleton k) | (k, vs) <- Map.assocs m, v <- Set.toList vs ] <> Map.fromSet (const mempty) (Map.keysSet m)

   42 -- | Find the direct list of things that are referencing this module.
   43 directDepsOn :: Graph -> Text -> Set Text
   44 directDepsOn = flip (Map.findWithDefault mempty) . unGraph

   46 -- | Flood fill to find all transitive dependencies on a starting module, excluding the original key.
   47 allDepsOn :: Graph -> Text -> Set Text
   48 allDepsOn = flood . directDepsOn

   50 -- | Find an example path between two modules.
   51 --
   52 -- This is a short path, but the important part is that it represents how connectivy works.
   53 why :: Graph -> Text -> Text -> [Text]
   54 why m from to = fromMaybe [] $ findFirst bfsOn head (\ks -> (:ks) <$> (Set.toList . directDepsOn m . head) ks) [from] ((== to) . head)

   56 -- | Find all paths between two modules as a restricted graph of the intersection of
   57 -- reachable nodes from the start and to the end.
   58 allPathsTo :: Graph -> Text -> Text -> Graph
   59 allPathsTo m from to = restrictTo m $ allDepsOn m from `Set.intersection` allDepsOn (reverseEdges m) to

   61 -- | Count the number of transitive dependencies for each module.
   62 rankings :: Graph -> Map Text Int
   63 rankings g = Map.fromList $ mapMaybeWithKey (Just . length . allDepsOn g) g

   65 -- | Visit each node in the graph and apply a function to it.
   66 mapMaybeWithKey :: NFData a => (Text -> Maybe a) -> Graph -> [(Text, a)]
   67 mapMaybeWithKey f = mapMaybe sequenceA . parMap rdeepseq (ap (,) f) . Map.keys . unGraph

   69 -- | Restrict a graph to only the given set of modules.
   70 restrictTo :: Graph -> Set Text -> Graph
   71 restrictTo (Graph m) keep = Graph . flip Map.mapMaybeWithKey m $ \k' v ->
   72   (if Set.notMember k' keep then Nothing else nonNullSet (Set.intersection keep v))
   73     where
   74         nonNullSet v = if Set.null v then Nothing else Just v

   76 -- | Convert a Graph to a Tree. If there is a cycle, treat the cycle point as a leaf.
   77 graphToTree :: Text -> Graph -> Tree Text
   78 graphToTree startingAt (Graph rels) = go mempty startingAt
   79   where
   80     go seen node =
   81       let children = Set.toList $ Map.findWithDefault mempty node rels
   82       in Tree.Node node $ if (Set.member node seen) then [] else fmap (go (Set.insert node seen)) children
