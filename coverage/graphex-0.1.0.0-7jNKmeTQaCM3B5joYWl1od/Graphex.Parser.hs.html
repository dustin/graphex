    6 -- | Implements a very crude Haskell import parser.
    7 --
    8 -- All it does is parse full module names that are imported in a file,
    9 -- ignoring false positives.
   10 --
   11 -- It does not use CPP, so all conditional imports are parsed out. For
   12 -- graphex, this makes sense in a way - those are all still dependencies.
   13 module Graphex.Parser where
   14 
   15 import Data.Text qualified as T
   16 import Data.Void
   17 import Data.String (IsString)
   18 import Data.Maybe (isJust)
   19 import Data.Either (rights)
   20 
   21 import Text.Megaparsec
   22 import Text.Megaparsec.Char
   23 import Replace.Megaparsec (sepCap, streamEdit)
   24 
   25 import Graphex.Core
   26 
   27 importParser
   28   :: MonadParsec e s m
   29   => Token s ~ Char
   30   => IsString (Tokens s)
   31   => m Import
   32 importParser = do
   33   -- This newline helps us avoid the word "import" in comments or identifiers.
   34   -- Basically, we insist imports are at the beginning of a line (which Haskell does
   35   -- too I think)
   36   _ <- newline
   37   _ <- string "import"
   38   space1
   39   -- Handle prefix qualified imports
   40   _ <- isJust <$> optional (string "qualified")
   41   space
   42   -- Handle PackageImports
   43   pkg <- optional $ between (char '"') (char '"') $ some $ alphaNumChar <|> char '-' <|> char '_'
   44   space
   45 
   46   modid <- some $ alphaNumChar <|> char '.' <|> char '_'
   47 
   48   pure Import
   49     { module_ = ModuleName $ T.pack modid
   50     , package = T.pack <$> pkg
   51     }
   52 
   53 importsParser
   54   :: forall e s m
   55    . MonadParsec e s m
   56   => Token s ~ Char
   57   => IsString (Tokens s)
   58   => m [Import]
   59 importsParser = rights <$> sepCap importParser
   60 
   61 removeBlockComments :: String -> String
   62 removeBlockComments = streamEdit (blockCommentParser @Void) (\_ -> "")
   63 
   64 blockCommentParser
   65   :: forall e s m
   66    . MonadParsec e s m
   67   => Token s ~ Char
   68   => IsString (Tokens s)
   69   => m String
   70 blockCommentParser = do
   71   _ <- string "{-"
   72   manyTill anySingle (string "-}")
   73   
   74 parseFileImports :: FilePath -> IO [Import]
   75 parseFileImports fp = do
   76   contents <- removeBlockComments <$> readFile fp
   77   case runParser (importsParser @Void) fp contents of
   78     Left err -> error $ unlines $ [mconcat ["Failed to parse ", fp, ":"], errorBundlePretty err]
   79     Right x -> pure x
