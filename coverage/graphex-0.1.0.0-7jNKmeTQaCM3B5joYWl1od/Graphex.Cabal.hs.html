   57 discoverCabalModules :: FilePath -> IO [Module]
   58 discoverCabalModules cabalFile = do
   59   gpd <- readGenericPackageDescription silent cabalFile
   60   let PackageDescription{..} = flattenPackageDescription gpd
   61   let candidateModules = mconcat
   62         [ do
   63             Library{..} <- maybeToList library
   64             srcDir <- hsSourceDirs libBuildInfo
   65             exMod <- exposedModules
   66             pure Module
   67               { name = fromString $ mconcat $ intersperse "." $ Cabal.components exMod
   68               , path = sourceDirToFilePath srcDir </> Cabal.toFilePath exMod <.> ".hs"
   69               }
   70         ] -- TODO: exes + other-modules
   71 
   72   filterM (doesFileExist . (.path)) candidateModules
   73 
   74 discoverCabalModuleGraph :: IO ModuleGraph
   75 discoverCabalModuleGraph = do
   76   fs <- getDirectoryContents "." -- XXX
   77   mods <- fmap fold . traverse discoverCabalModules . filter ((".cabal" ==) . takeExtension) $ fs
   78 
   79   let modSet = foldMap (Set.singleton . name) mods
   80   gs <- for mods $ \Module{..} -> do
   81     internalImps <- filter (\Import{..} -> Set.member module_ modSet) <$> parseFileImports path
   82     pure $ mkModuleGraph name $ fmap module_ internalImps
   83   pure $ fold gs
