   12 newtype Graph a = Graph { unGraph :: Map a (Set a) }
   13   deriving stock Eq
   14 
   15 instance Ord a => Semigroup (Graph a) where
   16   Graph x <> Graph y = Graph (Map.unionWith (<>) x y)

   18 instance Ord a => Monoid (Graph a) where
   19   mempty = Graph mempty

   21 singletonGraph :: Ord a => a -> a -> Graph a
