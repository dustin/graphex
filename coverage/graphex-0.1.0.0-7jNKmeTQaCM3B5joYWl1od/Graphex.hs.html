   30 -- | Convert a dependency file to a graph.
   31 depToGraph :: GraphDef -> Graph Text
   32 depToGraph GraphDef{..} = Graph (links <> allNodes)
   33     where
   34         links = Map.fromListWith (<>) [ (name from, Set.singleton (name to)) | Edge{..} <- edges]
   35         allNodes = Map.fromList [(label, mempty) | Node{..} <- Map.elems nodes]
   36         name = (.label) . (nodes Map.!) :: Text -> Text
   37 
   38 -- | Convert a graph back to our dependency file format.
   39 graphToDep :: Graph Text -> GraphDef
   40 graphToDep (Graph m) = GraphDef {
   41     title = "Internal Package Dependencies",
   42     edges = [ Edge { from = k, to = v } | (k, vs) <- Map.assocs m, v <- Set.toList vs ],
   43     nodes = Map.fromList [ (k, Node k Nothing) | k <- Map.keys m ]
   44     }

   46 -- | Reverse all the arrows in the graphs.
   47 reverseEdges :: Ord a => Graph a -> Graph a
   48 reverseEdges (Graph m) = Graph $ Map.fromListWith (<>) [ (v, Set.singleton k) | (k, vs) <- Map.assocs m, v <- Set.toList vs ] <> Map.fromSet (const mempty) (Map.keysSet m)

   50 -- | Find the direct list of things that are referencing this module.
   51 directDepsOn :: Ord a => Graph a -> a -> Set a
   52 directDepsOn = flip (Map.findWithDefault mempty) . unGraph
   53 
   54 -- | Flood fill to find all transitive dependencies on a starting module, excluding the original key.
   55 allDepsOn :: Ord a => Graph a -> a -> Set a
   56 allDepsOn = flood . directDepsOn
   57 
   58 -- | Find an example path between two modules.
   59 --
   60 -- This is a short path, but the important part is that it represents how connectivy works.
   61 why :: Ord a => Graph a -> a -> a -> [a]
   62 why m from to = fromMaybe [] $ findFirst bfsOn head (\ks -> (:ks) <$> (Set.toList . directDepsOn m . head) ks) [from] ((== to) . head)
   63 
   64 -- | Find all paths between two modules as a restricted graph of the intersection of
   65 -- reachable nodes from the start and to the end.
   66 allPathsTo :: Ord a => Graph a -> a -> a -> Graph a
   67 allPathsTo m from to = restrictTo m $ allDepsOn m from `Set.intersection` allDepsOn (reverseEdges m) to
   68 
   69 -- | Count the number of transitive dependencies for each module.
   70 rankings :: (NFData a, Ord a) => Graph a -> [(Int, a)]
   71 rankings g = sortOn (first Down) . fmap swap $ mapMaybeWithKey (Just . length . allDepsOn g) g
   72 
   73 -- | Visit each node in the graph and apply a function to it.
   74 mapMaybeWithKey :: (NFData a, NFData g) => (g -> Maybe a) -> Graph g -> [(g, a)]
   75 mapMaybeWithKey f = mapMaybe sequenceA . parMap rdeepseq (ap (,) f) . Map.keys . unGraph
   76 
   77 -- | Restrict a graph to only the given set of modules.
   78 restrictTo :: Ord a => Graph a -> Set a -> Graph a
   79 restrictTo (Graph m) keep = Graph . flip Map.mapMaybeWithKey m $ \k' v ->
   80   if Set.notMember k' keep then Nothing else Just (Set.intersection keep v)
   81 
   82 -- | Convert a Graph to a Tree. If there is a cycle, treat the cycle point as a leaf.
   83 graphToTree :: Ord a => a -> Graph a -> Tree a
   84 graphToTree startingAt (Graph rels) = go mempty startingAt
   85   where
   86     go seen node =
   87       let children = Set.toList $ Map.findWithDefault mempty node rels
   88       in Tree.Node node $ if Set.member node seen then [] else fmap (go (Set.insert node seen)) children
   89 
   90 -- | Find the longest path in a Graph.
   91 longest :: (NFData a, Ord a) => Graph a -> [a]
   92 longest g = reverse . snd . maximumOn (length . snd) $ mapMaybeWithKey one g
   93   where
   94     one = Just . fmap fst . maximumOn length . dfsWith store seen (\ks -> (:ks) . (, succ . snd . head $ ks) <$> foldMap (Set.toList . directDepsOn g . fst) (listToMaybe ks)) . (:[]) . (,1::Int)
   95     -- store and head cannot be called with empty lists.
   96     store (head -> (k,l)) = Map.insert k l
   97     seen (head -> (k,l)) = maybe False (< l) . Map.lookup k
   98     maximumOn :: Ord b => (a -> b) -> [a] -> a
   99     maximumOn f = maximumBy (comparing f)
