   44 toLookingGlass
   45   :: Text -- ^ title
   46   -> Map ModuleName Color -- ^ Optionally re-color any nodes (black default)
   47   -> ModuleGraph
   48   -> GraphDef
   49 toLookingGlass title colors Graph{..} =
   50   let mkNodeId (ModuleName m) = m
   51       mkNode m =
   52         ( mkNodeId m
   53         , Node
   54           { color = unColor <$> Map.lookup m colors
   55           , label = mkNodeId m
   56           }
   57         )
   58   in GraphDef
   59      { nodes = Map.fromList $ fmap (\(m, _) -> mkNode m) $ Map.toList unGraph
   60      , edges = Map.toList unGraph >>= \(m, children) -> fmap (\c -> Edge{from = mkNodeId c, to = mkNodeId m}) $ Set.toList children
   61      , ..
   62      }
