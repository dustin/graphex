   89 discoverCabalModules :: CabalDiscoverOpts -> FilePath -> IO [Module]
   90 discoverCabalModules CabalDiscoverOpts{..} cabalFile = do
   91   gpd <- readGenericPackageDescription silent cabalFile
   92   let PackageDescription{..} = flattenPackageDescription gpd
   93   let candidateModules = mconcat
   94         [ do
   95             Library{..} <- mconcat [maybeToList library, subLibraries]
   96             srcDir <- hsSourceDirs libBuildInfo
   97             exMod <- exposedModules
   98             let name = unUnqualComponentName <$> libraryNameString libName
   99             guard $ Discovered == foldMap1 (`discoversUnit` CabalLibraryUnit name) toDiscover
  100             pure Module
  101               { name = fromString $ mconcat $ intersperse "." $ Cabal.components exMod
  102               , path = if exMod `elem` libBuildInfo.autogenModules then ModuleNoFile else ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath exMod <.> ".hs"
  103               }
  104         , do
  105             Executable{..} <- executables
  106             srcDir <- hsSourceDirs buildInfo
  107             otherMod <- "Main" : buildInfo.otherModules
  108             guard $ Discovered == foldMap1 (`discoversUnit` CabalExecutableUnit (unUnqualComponentName exeName)) toDiscover
  109             pure Module
  110               { name = fromString $
  111                 if otherMod == "Main"
  112                 then unUnqualComponentName exeName ++ "-Main"
  113                 else mconcat $ intersperse "." $ Cabal.components otherMod
  114               , path = if otherMod `elem` buildInfo.autogenModules then ModuleNoFile else ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath otherMod <.> ".hs"
  115               }
  116         , do
  117             TestSuite{..} <- testSuites
  118             srcDir <- hsSourceDirs testBuildInfo
  119             otherMod <- testBuildInfo.otherModules
  120             guard $ Discovered == foldMap1 (`discoversUnit` CabalTestsUnit (unUnqualComponentName testName)) toDiscover
  121 
  122             pure Module
  123               { name = fromString $ mconcat $ intersperse "." $ Cabal.components otherMod
  124               , path = if otherMod `elem` testBuildInfo.autogenModules then ModuleNoFile else ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath otherMod <.> ".hs"
  125               }
  126         ]
  127 
  128   catMaybes <$> pooledMapConcurrentlyN numJobs validateModulePath candidateModules
  129 
  130 validateModulePath :: Module -> IO (Maybe Module)
  131 validateModulePath m = do
  132   let mkModule p = Module m.name p
  133   case m.path of
  134     ModuleFile fp -> do
  135       fileExists <- doesFileExist fp
  136       -- If a discovered module has a filepath but doesn't exist, it means
  137       -- that the unit had multiple source directories. We assume it will
  138       -- exist in one of those directories.
  139       pure $ if fileExists then Just $ mkModule $ ModuleFile fp else Nothing
  140     ModuleNoFile -> pure $ Just $ mkModule ModuleNoFile

  142 data CabalUnitType =
  143     CabalLibrary
  144   | CabalExecutable
  145   | CabalTests
  146   deriving stock (Show, Eq, Ord)
  147 
  148 data CabalUnit =
  149     CabalLibraryUnit (Maybe String)
  150   | CabalExecutableUnit String
  151   | CabalTestsUnit String
  152   deriving stock (Show, Eq, Ord)
  153 
  154 data CabalDiscoverType =
  155     CabalDiscoverAll CabalUnitType
  156   | CabalDiscover CabalUnit
  157   | CabalDontDiscoverAll CabalUnitType
  158   | CabalDontDiscover CabalUnit
  159   deriving stock (Show, Eq, Ord)
  160 
  161 data Discovery =
  162     Discovered
  163   | Hidden
  164   | Passed
  165   deriving stock (Show, Eq, Ord)
  166 
  167 instance Semigroup Discovery where
  168   _ <> Discovered = Discovered
  169   _ <> Hidden     = Hidden
  170   x <> Passed     = x
  171 
  172 flipDiscover :: Discovery -> Discovery
  173 flipDiscover = \case
  174   Passed -> Passed
  175   Hidden -> Discovered
  176   Discovered -> Hidden
  177 
  178 discoversUnit :: CabalDiscoverType -> CabalUnit -> Discovery
  179 discoversUnit = curry $ \case
  180   (CabalDiscoverAll CabalLibrary, CabalLibraryUnit{}) -> Discovered
  181   (CabalDiscoverAll CabalLibrary, _) -> Passed
  182   (CabalDiscoverAll CabalExecutable, CabalExecutableUnit{}) -> Discovered
  183   (CabalDiscoverAll CabalExecutable, _) -> Passed
  184   (CabalDiscoverAll CabalTests, CabalTestsUnit{}) -> Discovered
  185   (CabalDiscoverAll CabalTests, _) -> Passed
  186   (CabalDiscover u1, u2) -> if u1 == u2 then Discovered else Passed
  187   (CabalDontDiscoverAll ty, u) -> flipDiscover $ CabalDiscoverAll ty `discoversUnit` u
  188   (CabalDontDiscover u1, u2) -> flipDiscover $ CabalDiscover u1 `discoversUnit` u2

  197 data CabalGraph = CabalGraph

  200   }

  202 discoverCabalModuleGraph :: CabalDiscoverOpts -> IO CabalGraph
  203 discoverCabalModuleGraph opts@CabalDiscoverOpts{..} = do
  204   fs <- getDirectoryContents "."
  205   mods <- fmap fold . traverse (discoverCabalModules opts) . filter ((".cabal" ==) . takeExtension) $ fs
  206 
  207   let modMap = foldMap (\m@Module{..} -> Map.singleton name m) mods
  208   mg <- case pruneTo of
  209     Nothing -> do
  210       gs <- pooledForConcurrentlyN numJobs mods $ \Module{..} -> case path of
  211         ModuleFile modPath -> do
  212           allImps <- parseFileImports modPath
  213           let filteredImps = filter (\Import{..} -> Map.member module_ modMap) allImps
  214           pure $ mkModuleGraph name $ fmap module_ (if includeExternal then allImps else filteredImps)
  215         ModuleNoFile -> pure $ mkModuleGraph name mempty
  216       pure $ fold gs
  217     Just keep -> do
  218       sem <- newQSem numJobs
  219       graphRef <- newTVarIO @ModuleGraph mempty
  220       seenRef <- newTVarIO @(Set.Set ModuleName) mempty
  221       let go Module{..} = case path of
  222             ModuleFile modPath -> do
  223               importedMods <- bracket_ (waitQSem sem) (signalQSem sem) $ do
  224                 seen <- (Set.member name) <$> readTVarIO seenRef
  225                 if seen then pure [] else do
  226                   logit $ unwords ["Start parsing imports for", T.unpack $ unModuleName name]
  227                   allImps <- parseFileImports modPath
  228                   logit $ unwords ["Finished parsing imports for", T.unpack $ unModuleName name]
  229                   let filteredImps = filter (\Import{..} -> Map.member module_ modMap) allImps
  230                   atomically $ modifyTVar' graphRef (\g -> mappend g $ mkModuleGraph name $ fmap module_ (if includeExternal then allImps else filteredImps))
  231                   atomically $ modifyTVar' seenRef (Set.insert name)
  232                   pure $ mapMaybe (\Import{..} -> Map.lookup module_ modMap) allImps
  233               mapConcurrently_ go importedMods
  234             ModuleNoFile -> pure ()
  235       let pruneToMods = fmap snd $ filter (keep . fst) (Map.toList modMap)
  236       logit $ unwords ["Pruning graph to transitive imports of:", show $ fmap (unModuleName . name) pruneToMods]
  237       mapConcurrently_ go pruneToMods
  238       readTVarIO graphRef
  239   pure CabalGraph
  240     { moduleGraph = mg
  241     , modulesByName = modMap
  242     }

  244 mkCabalFileGraph :: CabalGraph -> Graph FilePath
  245 mkCabalFileGraph CabalGraph{..} =
  246     Graph
  247     { unGraph = Map.fromList $ flip mapMaybe (Map.toList $ unGraph moduleGraph) $ \(p, cs) -> do
  248         p' <- Map.lookup p modulesByName >>= moduleFilePath
  249         let cs' = Set.fromList $ mapMaybe (\c -> Map.lookup c modulesByName >>= moduleFilePath) $ Set.toList cs
  250         Just (p', cs')
  251     , attributes = Map.fromList $ flip mapMaybe (Map.toList $ attributes moduleGraph) $ \(n, as) -> do
  252         n' <- Map.lookup n modulesByName >>= moduleFilePath
  253         Just (n', as)
  254     }
