   59 discoverCabalModules :: CabalDiscoverOpts -> FilePath -> IO [Module]
   60 discoverCabalModules CabalDiscoverOpts{..} cabalFile = do
   61   gpd <- readGenericPackageDescription silent cabalFile
   62   let PackageDescription{..} = flattenPackageDescription gpd
   63   let candidateModules = mconcat
   64         [ do
   65             guard $ Set.member CabalLibraries toDiscover
   66             Library{..} <- mconcat [maybeToList library, subLibraries]
   67             srcDir <- hsSourceDirs libBuildInfo
   68             exMod <- exposedModules
   69             pure Module
   70               { name = fromString $ mconcat $ intersperse "." $ Cabal.components exMod
   71               , path = ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath exMod <.> ".hs"
   72               }
   73         , do
   74             guard $ Set.member CabalExecutables toDiscover
   75             Executable{..} <- executables
   76             srcDir <- hsSourceDirs buildInfo
   77             otherMod <- "Main" : buildInfo.otherModules
   78             pure Module
   79               { name = fromString $
   80                 if otherMod == "Main"
   81                 then unUnqualComponentName exeName ++ "-Main"
   82                 else mconcat $ intersperse "." $ Cabal.components otherMod
   83               , path = ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath otherMod <.> ".hs"
   84               }
   85         , do
   86             guard $ Set.member CabalTests toDiscover
   87             TestSuite{..} <- testSuites
   88             srcDir <- hsSourceDirs testBuildInfo
   89             otherMod <- testBuildInfo.otherModules
   90             pure Module
   91               { name = fromString $ mconcat $ intersperse "." $ Cabal.components otherMod
   92               , path = ModuleFile $ sourceDirToFilePath srcDir </> Cabal.toFilePath otherMod <.> ".hs"
   93               }
   94         ]
   95 
   96   pooledMapConcurrently validateModulePath candidateModules
   97 
   98 validateModulePath :: Module -> IO Module
   99 validateModulePath m = do
  100   path <- case m.path of
  101     ModuleFile fp -> do
  102       fileExists <- doesFileExist fp
  103       pure $ if fileExists then ModuleFile fp else ModuleNoFile
  104     ModuleNoFile -> pure ModuleNoFile
  105   pure Module {name = m.name, path = path}

  107 data CabalModuleType =
  108   CabalLibraries
  109   | CabalExecutables
  110   | CabalTests
  111   deriving stock (Show, Eq, Ord)
  112 
  113 data CabalDiscoverOpts = CabalDiscoverOpts
  114   { toDiscover      :: Set CabalModuleType
  115   , includeExternal :: Bool
  116   } deriving stock (Show, Eq)
  117 
  118 discoverCabalModuleGraph :: CabalDiscoverOpts -> IO ModuleGraph
  119 discoverCabalModuleGraph opts@CabalDiscoverOpts{..} = do
  120   fs <- getDirectoryContents "." -- XXX
  121   mods <- fmap fold . pooledMapConcurrently (discoverCabalModules opts) . filter ((".cabal" ==) . takeExtension) $ fs
  122 
  123   let modSet = foldMap (Set.singleton . name) mods
  124   gs <- pooledForConcurrently mods $ \Module{..} -> case path of
  125     ModuleFile modPath -> do
  126       allImps <- parseFileImports modPath
  127       let filteredImps =
  128             if includeExternal
  129             then allImps
  130             else filter (\Import{..} -> Set.member module_ modSet) allImps
  131       pure $ mkModuleGraph name $ fmap module_ filteredImps
  132     ModuleNoFile -> pure $ mkModuleGraph name mempty
  133   pure $ fold gs
